# 快速开始

本节将介绍如何在项目中使用 Dinert Element UI

# 完整引入

如果你对打包后的文件大小不是很在乎，那么使用完整导入会更方便。

main.js

// main.js
import Vue from 'vue'
import '@dinert/element-ui/style'
import DinertComponents from '@dinert/element-ui'
import ElementUI from 'element-ui'


Vue.use(ElementUI)
Vue.use(DinertComponents)

<template>
