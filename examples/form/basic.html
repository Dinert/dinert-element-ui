export default {
    data() {
        return {
            form: {
                labelWidth: 'auto',
                required: true,
                showLabel(model) {
                    return true
                },

                    name: {
                        label: '必填',
                        type: 'input',
                        required: true,
                        showLabel: false,

                        options: {

                        }

                        required: true,
                        showLabel: false,
                        options: {
                            options: [
                                {label: '显示当我的长度过长长长长长长', value: true},
                                {label: '隐藏', value: false},
                            ]
                        }
                    },
                    status: {
                        label: '选择',
                        type: 'select-v2',
                        showLabel(model) {
                            return model.name2
                        },
                        options: {
                            options: [
                                {label: '显示当我的长度过长长长长长长', value: true},

                            ]
                        }
                    }
                },
                model: {

                }
            },
        }
    },
    methods: {
        save() {

            this.$refs.formRef.$refs.formRef.validate((valid, obj) => {
                if (valid) {
                    console.log('保存', obj)

                    console.log('校验失败', obj)
                }
            })
        },

            this.$refs.formRef.$refs.formRef.resetFields()
        }
    }
}
